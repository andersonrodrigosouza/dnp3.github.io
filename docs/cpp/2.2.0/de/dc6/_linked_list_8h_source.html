341 template <class ValueType, class IndexType>
342 void LinkedList<ValueType, IndexType>::Remove(ListNode<ValueType>* apNode)
343 {
344  if(apNode->prev == nullptr) // it's the head
345  {
346  if (apNode->next == nullptr)
347  {
348  pHead = pTail = nullptr; // list is now empty
349  }
350  else
351  {
352  pHead = apNode->next; // head but not tail
353  }
354  }
355  else
356  {
357  if(apNode->next == nullptr) pTail = apNode->prev; // was only the tail
358  }
359 
360  // attach the adjacent nodes to eachother if they exist
361  Link(apNode->prev, apNode->next);
362 
363  // Now that the data list is complete, attach the freed node to the front of the free list
364  apNode->next = pFree;
365  if(pFree != nullptr) pFree->prev = apNode;
366  apNode->prev = nullptr; // it's the head now
367  pFree = apNode;
368  --(this->size);
369 }
